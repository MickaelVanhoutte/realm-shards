export const getTile = (map: MapData, x: number, y: number): TileType | null => {
    if (x < 0 || x >= map.width || y < 0 || y >= map.height) {
        return null;
    }
    const index = y * map.width + x;
    return map.tiles[index];
};
 
export const getTileInfo = (tileType: TileType): TileInfo => {
